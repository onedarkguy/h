File 2: app.js 
// Function to fetch and display JSON
async function fetchJSON() {
    try {
        const response = await fetch('data.json');
        if (!response.ok) throw new Error("Network response was not ok");
        const data = await response.json();
        
        renderTable(data, ['Name', 'Age', 'City'], ['name', 'age', 'city']);
    } catch (error) {
        console.error("Error fetching JSON:", error);
    }
}

// Function to fetch and display XML
async function fetchXML() {
    try {
        const response = await fetch('data.xml');
        const text = await response.text();
        const parser = new DOMParser();
        const xmlDoc = parser.parseFromString(text, "text/xml");
        
        const students = xmlDoc.getElementsByTagName("student");
        let dataList = [];

        for (let i = 0; i < students.length; i++) {
            dataList.push({
                name: students[i].getElementsByTagName("name")[0].childNodes[0].nodeValue,
                age: students[i].getElementsByTagName("age")[0].childNodes[0].nodeValue
            });
        }
        
        renderTable(dataList, ['Name', 'Age'], ['name', 'age']);
    } catch (error) {
        console.error("Error fetching XML:", error);
    }
}

// Helper function to render the table dynamically
function renderTable(data, headers, keys) {
    const head = document.getElementById('table-head');
    const body = document.getElementById('table-body');
    
    // Clear existing content
    head.innerHTML = "";
    body.innerHTML = "";

    // Create Headers
    headers.forEach(text => {
        let th = document.createElement('th');
        th.innerText = text;
        head.appendChild(th);
    });

    // Create Rows
    data.forEach(item => {
        let row = body.insertRow();
        keys.forEach(key => {
            let cell = row.insertCell();
            cell.innerText = item[key];
        });
    });
}
