import { useEffect, useState } from "react";

function Weather() {

  const [temperature, setTemperature] = useState("");
  const [condition, setCondition] = useState("");

  useEffect(() => {

    fetch("https://api.openweathermap.org/data/2.5/weather?q=Mumbai&appid=a5f7d76e31bcf68c431ec52a6c23d75b&units=metric")
      .then((response) => response.json())
      .then((data) => {

        setTemperature(data.main.temp);
        setCondition(data.weather[0].main);

      });

  }, []);

  return (
    <div>
      <h2>Weather Information</h2>
      <p>Temperature: {temperature} °C</p>
      <p>Condition: {condition}</p>
    </div>
  );
}

export default Weather;

import { useEffect, useState } from "react";

function FetchUsers() {

  const [users, setUsers] = useState([]);

  useEffect(() => {

    fetch("https://jsonplaceholder.typicode.com/users")
      .then((response) => response.json())
      .then((data) => setUsers(data));

  }, []);

  return (
    <div>

      <h2>Users List (API Data)</h2>

      <ul>
        {users.map((user) => (
          <li key={user.id}>
            {user.name} - {user.email}
          </li>
        ))}
      </ul>

    </div>
  );
}

export default FetchUsers;

import { useState, useEffect } from "react";

function LocalStorageUser() {

  const [name, setName] = useState("");
  const [savedUser, setSavedUser] = useState(null);

  const saveUser = () => {

    const user = { name };

    localStorage.setItem("user", JSON.stringify(user));

    setSavedUser(user);
  };

  useEffect(() => {

    const storedUser = JSON.parse(localStorage.getItem("user"));

    if (storedUser) {
      setSavedUser(storedUser);
    }

  }, []);

  return (
    <div>

      <h2>Local Storage Example</h2>

      <input
        type="text"
        placeholder="Enter Name"
        onChange={(e) => setName(e.target.value)}
      />

      <button onClick={saveUser}>Save</button>

      {savedUser && (
        <p>Stored User: {savedUser.name}</p>
      )}

    </div>
  );
}

export default LocalStorageUser;
